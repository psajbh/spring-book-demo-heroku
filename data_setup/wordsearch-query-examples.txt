
select * from wordsearch 
where
common = '1'
and
word like '%a%'
and
word like '%e%'
and 
word like '%t%'
-- and
-- word like '%t%'
and
word not like '%u%'
and 
word not like '%d%'
and 
word not like '%i%'
and
word not like '%o%'
and
word not like '%m%'
and
word not like '%p%'
and
word not like '%y%'

and char1 != 'a'
and char1 != 'e'
;
